class WeatherStationParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.stations = []
        self.in_target_cell = False
        
    def handle_starttag(self, tag, attrs):
        # Convert attributes list to a dictionary for easy lookup
        attrs_dict = dict(attrs)
        
        # 1. Look for the element carrying the overlib JavaScript trigger
        if 'onmouseover' in attrs_dict and 'overlib' in attrs_dict['onmouseover']:
            js_call = attrs_dict['onmouseover']
            
            # Use safe Python string splits instead of regular expressions
            # Split by single quotes to isolate the string parameters inside overlib('...', CAPTION, '...')
            parts = js_call.split("'")

                station_name = parts[3].strip()
                inner_html = parts[1]

            self.in_target_cell = True

        if self.in_target_cell:
